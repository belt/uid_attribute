def uid_attribute(uid_attr = :uid)
  install_uid_attribute_validators(uid_attr)

  class_eval("class << self;attr_accessor :uid_attr;attr_accessor :uid_object; end")
  @uid_object = false
  protected
  @uid_attr = uid_attr
end
